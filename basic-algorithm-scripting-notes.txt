******************
 ARRAYS METHODS
******************

//"PUSH" OR APPEND DATA TO END OF ARRAY//
var arr = [1,2,3];
arr.push(4);
// arr is now [1,2,3,4]


//"POP" OFF VALUE OF THE LAST ITEM OF ARRAY THEN CAN STORE ASSIGN TO VARIABLE //
var oneDown = [1, 4, 6].pop();

//"SHIFT" OFF VALUE OF THE FIRST ITEM OF ARRAY THEN CAN STORE ASSIGN TO VARIABLE //
var ourArray = ["Stimpson", "J", ["cat"]];
removedFromOurArray = ourArray.shift();


//"UNSHIFT" ADD ELEMENT TO BEGINNING OF ARRAY THEN CAN STORE ASSIGN TO VARIABLE //
var ourArray = ["Stimpson", "J", "cat"];
ourArray.shift(); // ourArray now equals ["J", "cat"]
ourArray.unshift("Happy"); 
// ourArray now equals ["Happy", "J", "cat"]
